    <script>
        
const mangaId = 1;
const apiUrl = `https://api.jikan.moe/v4/manga/${mangaId}/full`;

fetch(apiUrl)
  .then(response => {
    if (!response.ok) {
      throw new Error(`HTTP error! Status: ${response.status}`);
    }
    return response.json();
  })
  .then(data => {
    const chapters = data.data.chapters;
    console.log(`Number of chapters: ${chapters}`);
    const mangaElement = document.querySelector("#manga");
    mangaElement.textContent = JSON.stringify(chapters);
    localStorage.setItem("manga", JSON.stringify(chapters));
    console.log(data);
  })
  .catch(error => {
    console.error(error);
  });
